def sendURL(url):
    with virustotal_python.Virustotal("c38901fa866932438ae97c33b0695c5cece261e980c08f078242e8876421de7e") as vtotal:
        if url in VTsearch:
            #print("Accessing cache to return the result")
            return Row('Out1','Out2','Out3','Out4')(VTsearch[url]["harmless"],VTsearch[url]["malicious"],VTsearch[url]["suspicious"],VTsearch[url]["undetected"])
        try:
            url_id = urlsafe_b64encode(url.encode()).decode().strip("=")
            #print(f"Sending URL: {url} to VirusTotal...")
            response = vtotal.request(f"urls/{url_id}")
            VTsearch[url]= response.data["attributes"]["last_analysis_stats"]
            return Row('Out1','Out2','Out3','Out4')(response.data["attributes"]["last_analysis_stats"]["harmless"],response.data["attributes"]["last_analysis_stats"]["malicious"],response.data["attributes"]["last_analysis_stats"]["suspicious"],response.data["attributes"]["last_analysis_stats"]["undetected"])
        except virustotal_python.VirustotalError as err:
            print(f"Failed to send URL: {url} for analysis and get the report: {err}")
            return {}
